104INLINE Vector3f Corrector_getAccCorrected(Corrector_t *corr, Vector3f *acc)
105{
106 Vector3f result =
107 {
108 (acc->x + corr->accCorrOffsetx) * corr->accCorrGainx,
109 (acc->y + corr->accCorrOffsety) * corr->accCorrGainy,
110 (acc->z + corr->accCorrOffsetz) * corr->accCorrGainz
111 };
112 return result;
113}
