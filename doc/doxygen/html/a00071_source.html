44INLINE float IMU_get_roll_quat(IMU_Status_t *mstatus)
45{
46 Quaternionf q = mstatus->orientation;
47 return atan2f(2.0f * (q.w * q.x + q.y * q.z), 1.0f - 2.0f * (q.x * q.x + q.y * q.y));
48}
