149/*
150 * @brief Stop motor in case pf panic.
151 * This function stops the motor by resetting PA4, PA6 and PA7
152 * which are connected to PWM, INA and INB of the motor driver.
153 * It doesn't disengage the clutch.
154 * It is meant to be called by fault exceptions handlers.
155 * It is INLINE in order to use no function call nor stack.
156 */
157//INLINE void MOTOR_stopPanic(void)
158__attribute__((always_inline)) inline void MOTOR_stopPanic(void)
159{
160 /* Reset PWN, INA and INB */
161 // LL_GPIO_ResetOutputPin(GPIOA,
162 // LL_GPIO_PIN_4 | LL_GPIO_PIN_6 | LL_GPIO_PIN_7);
163 /* Very low level : register acces */
164 GPIOA->BRR = LL_GPIO_PIN_4 | LL_GPIO_PIN_6 | LL_GPIO_PIN_7;
165}
